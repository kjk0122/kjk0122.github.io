```
public static List<Integer> findDuplicates(int[] nums) {
List<Integer> result = new ArrayList<>();
Set<Integer> set = new HashSet<>();
        for (int num : nums) {
            if (set.contains(num)) {
                result.add(num);
            } else {
                set.add(num);
            }
        }
        return result;
    }
```
